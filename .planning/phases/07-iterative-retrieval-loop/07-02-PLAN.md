       ```python
       def reformulate_query(
           self,
           *,
           original: str,
           prior_queries: list[str],
           paths: list[str],
           titles: list[str],
           reason: str,
       ) -> str:
           """Generate a substantively different query for a second retrieval attempt.

           Inputs intentionally exclude note content (D-06, PRIV-03). Returns first
           non-empty trimmed line of the LLM response (per RESEARCH code example).
           """
           prior_bulleted = (
               "\n".join(f"- {q}" for q in prior_queries) if prior_queries else "- (nenhuma)"
           )
           if paths:
               paths_titles_bulleted = "\n".join(
                   f"- {p}: {t}" for p, t in zip(paths, titles)
               )
           else:
               paths_titles_bulleted = "- (nenhuma)"
           prompt = REFORMULATION_PROMPT.format(
               original_query=original,
               prior_queries_bulleted=prior_bulleted,
               paths_titles_bulleted=paths_titles_bulleted,
               reason=reason,
           )
           raw = self._sync_fn(
               endpoint_url=self._endpoint_url,
               model_id=self._model_id,
               messages=[{"role": "user", "content": prompt}],
           )
           for line in raw.splitlines():
               stripped = line.strip()
               if stripped:
                   return stripped
           return ""

       def judge_sufficiency(
           self,
           *,
           query: str,
           paths: list[str],
           context_sample: str,
       ) -> bool:
           """Optional LLM second-opinion sufficiency judge (D-03, opt-in).

           Returns True if response indicates 'sim'; False if 'nao'/'não';
           True (conservative — avoid wasted retrieval) if ambiguous.
           """
           paths_bulleted = (
               "\n".join(f"- {p}" for p in paths) if paths else "- (nenhuma)"
           )
           prompt = SUFFICIENCY_JUDGE_PROMPT.format(
               query=query,
               paths_bulleted=paths_bulleted,
               sample_chars=len(context_sample),
               context_sample=context_sample,
           )
           raw = self._sync_fn(
               endpoint_url=self._endpoint_url,
               model_id=self._model_id,
               messages=[{"role": "user", "content": prompt}],
           )
           lowered = raw.strip().lower()
           # Ambiguity policy (negative wins on tie). RESEARCH "LLM judge"
           # explicitly recommends conservative: when both 'sim' and a negation
           # appear (e.g. "sim porque nao falta nada") OR when neither appears,
           # return False so the orchestrator triggers another retrieval. The
           # cost of one extra LLM-driven attempt is small; the cost of skipping
           # reformulation when evidence is genuinely thin is a bad answer.
           has_sim = "sim" in lowered
           has_neg = "nao" in lowered or "não" in lowered
           if has_sim and not has_neg:
               return True
           return False  # nao OR ambiguous OR both -> conservative False
       ```

For attempt 2+, the origin is harder to track without modifying `QMDSearchHit`; pass `hybrid_top_score=None` to skip the top-score check on subsequent attempts (RESEARCH "Open Questions" #1 punts this; we honor that punt). Document this decision in code comments.
      4. Run `judge_sufficiency_deterministic(merged, min_top_score, min_hits, min_context_chars, hybrid_top_score=...)` first. If verdict.ok and `iterative_retrieval_judge` enabled, run `llm.judge_sufficiency(...)` as second pass. If LLM says False, override verdict to thin with `reason="LLM julgou insuficiente"`. (RESEARCH "Sufficiency Signals" — judge runs only after deterministic passes.)
      5. Append `RetrievalAttempt(...)` to attempts list (with paths from merged.notes, NEVER snippets)
      6. If verdict.ok or n == max_attempts: return (merged, build_trace(...))
      7. Reformulate via `llm.reformulate_query(original=query, prior_queries=prior_queries, paths=[...], titles=[...], reason=verdict.reason)`. titles = path basenames (e.g. `path.rsplit("/", 1)[-1]`).
      8. Jaccard guard (Pitfall 2, RESEARCH Pattern 3): `_too_similar(new_query, prior_queries, threshold=0.7)`. If True, mutate the LAST attempt's record to `reformulation_skipped=True`, `reformulation_reason="rewrite muito similar (Jaccard >= 0.7)"`, return early (do NOT run another retrieval).
      9. Update `accumulated_hits` (the per-attempt hits to carry into next attempt's merge), set `current_query = new_query`, append to `prior_queries`, loop.

    Cross-attempt merge (D-11, RESEARCH "Cross-attempt dedup"):
    - Merge new attempt's `notes` with `accumulated_notes` from prior attempts
    - Deduplicate by path: prefer the NEWER (current) attempt's note when paths collide (per RESEARCH "Subtle gotcha: dedup _within_ origin first … prefer attempt-2 hits when paths collide"). Implementation: build a dict keyed by path, iterate accumulated FIRST then new, so new overwrites.
    - Reassemble context_text via `RetrievalService._assemble_context` (truncates at MAX_CONTEXT_CHARS=24000)
    - **Insufficient_evidence preservation (BLOCKER FIX, D-10 / RET-04):** the merged result must propagate the truly-empty signal. Compute `merged_insufficient = current.insufficient_evidence and len(merged_notes) == 0` — i.e., the merge is insufficient ONLY when THIS attempt was insufficient AND the accumulated set added nothing either. Return `RetrievalResult(ok=True, notes=tuple(merged_notes), context_text=..., insufficient_evidence=merged_insufficient)`.
    - Why: if attempt 1 returns empty (`insufficient_evidence=True`) and attempt 2 also returns empty, the merged result MUST stay `insufficient_evidence=True` so ChatSession routes to the existing INSUFFICIENT_EVIDENCE_MSG path. Returning `insufficient_evidence=False` on a 0-note merge would cause the LLM to be called with empty context (D-10 / RET-04 violation).

        if not new_tokens:

       """IterativeRetrieval — bounded loop wrapping RetrievalService for D-09 / D-11 semantics."""
       from __future__ import annotations

       import logging
       import time
       from dataclasses import replace
       from typing import Callable

       from aurora.llm.service import LLMService
       from aurora.retrieval.contracts import (
           RetrievalAttempt,
           RetrievalResult,
           RetrievalTrace,
           RetrievedNote,
           SufficiencyVerdict,
       )
       from aurora.retrieval.service import RetrievalService, judge_sufficiency_deterministic
       from aurora.runtime.settings import RuntimeSettings, load_settings

       logger = logging.getLogger(__name__)

       RetrieveFn = Callable[[str, str, list[str]], RetrievalResult]

       _JUDGE_CONTEXT_SAMPLE_CHARS = 1000  # passed to LLM judge per RESEARCH


       def _too_similar(new: str, prior: list[str], *, threshold: float = 0.7) -> bool:
           """Token-Jaccard guard against synonym-swap reformulations (Pitfall 2, RESEARCH Pattern 3)."""
           new_tokens = set(new.lower().split())
           if not new_tokens:
               return True
           for q in prior:
               prior_tokens = set(q.lower().split())
               union = new_tokens | prior_tokens
               if not union:
                   continue
               overlap = len(new_tokens & prior_tokens) / len(union)
               if overlap >= threshold:
                   return True
           return False


       class IterativeRetrieval:
           """Bounded retrieval loop wrapping RetrievalService.

           Carry-forward composition: the caller (ChatSession) is responsible
           for applying carry-forward to first_attempt BEFORE calling run().
           Subsequent attempts go through retrieve_fn directly with NO
           carry-forward applied — this is intentional (Pitfall 1 in RESEARCH:
           re-applying carry-forward on attempt 2 would double-count prior
           supplements as new evidence).
           """

           def __init__(
               self,
               *,
               retrieval: RetrievalService,
               llm: LLMService,
               settings_loader: Callable[[], RuntimeSettings] = load_settings,
           ) -> None:
               s = settings_loader()
               self._retrieval = retrieval
               self._llm = llm
               self._enabled = s.iterative_retrieval_enabled
               self._max_attempts = s.retrieval_max_attempts
               self._min_top_score = s.retrieval_min_top_score
               self._min_hits = s.retrieval_min_hits
               self._min_context_chars = s.retrieval_min_context_chars
               self._judge_enabled = s.iterative_retrieval_judge

           def run(
               self,
               query: str,
               *,
               retrieve_fn: RetrieveFn,
               strategy: str,
               terms: list[str],
               intent: str,
               first_attempt: RetrievalResult | None = None,
           ) -> tuple[RetrievalResult, RetrievalTrace]:
               t_start = time.monotonic()

               # Disable path (D-18): single-shot, byte-equivalent semantics
               if not self._enabled:
                   result = first_attempt if first_attempt is not None else retrieve_fn(query, strategy, terms)
                   return result, RetrievalTrace(
                       enabled=False, intent=intent, attempts=(),
                       final_verdict_ok=not result.insufficient_evidence,
                       judge_used=False,
                       total_duration_ms=(time.monotonic() - t_start) * 1000.0,
                   )

               attempts: list[RetrievalAttempt] = []
               prior_queries: list[str] = [query]
               accumulated_notes: list[RetrievedNote] = []
               current_query = query
               current_result: RetrievalResult | None = first_attempt
               last_verdict: SufficiencyVerdict | None = None
               merged: RetrievalResult | None = None

               for n in range(1, self._max_attempts + 1):
                   t_attempt = time.monotonic()
                   if current_result is None:
                       current_result = retrieve_fn(current_query, strategy, terms)

                   # Cross-attempt merge (D-11): prefer newer attempt for path collisions
                   if n == 1:
                       merged = current_result
                   else:
                       merged = self._merge(current_result, accumulated_notes)

                   # Mode-aware top-score (RESEARCH finding 3 + Open Questions #1):
                   # On attempt 1 we trust the result's top score is hybrid-origin
                   # (RetrievalService default path is hybrid). On attempt 2+ we
                   # cannot tell origin without modifying QMDSearchHit, so we skip
                   # the top-score check by passing None.
                   if n == 1 and merged.notes:
                       hybrid_top_score: float | None = max(n_.score for n_ in merged.notes)
                   else:
                       hybrid_top_score = None

                   verdict = judge_sufficiency_deterministic(
                       merged,
                       min_top_score=self._min_top_score,
                       min_hits=self._min_hits,
                       min_context_chars=self._min_context_chars,
                       hybrid_top_score=hybrid_top_score,
                   )

                   # Optional LLM judge (D-03) — runs only when deterministic says ok
                   if verdict.ok and self._judge_enabled:
                       sample = merged.context_text[:_JUDGE_CONTEXT_SAMPLE_CHARS]
                       paths = [note.path for note in merged.notes]
                       if not self._llm.judge_sufficiency(
                           query=current_query, paths=paths, context_sample=sample,
                       ):
                           verdict = SufficiencyVerdict(
                               ok=False,
                               reason="LLM julgou insuficiente",
                               signals=("llm_judge=nao",),
                           )

                   last_verdict = verdict
                   attempts.append(
                       RetrievalAttempt(
                           attempt=n,
                           query=current_query,
                           strategy=strategy,
                           hit_count=len(merged.notes),
                           top_score=max((nn.score for nn in merged.notes), default=0.0),
                           paths=tuple(nn.path for nn in merged.notes),
                           verdict_ok=verdict.ok,
                           verdict_reason=verdict.reason,
                           verdict_signals=verdict.signals,
                           duration_ms=(time.monotonic() - t_attempt) * 1000.0,
                       )
                   )

                   if verdict.ok or n == self._max_attempts:
                       return merged, self._build_trace(intent, attempts, verdict.ok, t_start)

                   # Reformulate
                   new_query = self._llm.reformulate_query(
                       original=query,
                       prior_queries=prior_queries,
                       paths=[note.path for note in merged.notes],
                       titles=[note.path.rsplit("/", 1)[-1] for note in merged.notes],
                       reason=verdict.reason,
                   )
                   if _too_similar(new_query, prior_queries):
                       attempts[-1] = replace(
                           attempts[-1],
                           reformulation_skipped=True,
                           reformulation_reason="rewrite muito similar (Jaccard >= 0.7)",
                       )
                       return merged, self._build_trace(intent, attempts, verdict.ok, t_start)

                   attempts[-1] = replace(
                       attempts[-1],
                       reformulation_reason=verdict.reason,
                   )
                   accumulated_notes = list(merged.notes)
                   current_query = new_query
                   prior_queries.append(new_query)
                   current_result = None  # force fresh retrieve next iteration

               # Defensive return — loop should always exit via the verdict.ok / max-attempts branch
               assert merged is not None and last_verdict is not None
               return merged, self._build_trace(intent, attempts, last_verdict.ok, t_start)

           def _merge(
               self, current: RetrievalResult, accumulated: list[RetrievedNote],
           ) -> RetrievalResult:
               """Cross-attempt merge: prefer newer attempt for path collisions (D-11).

               Insufficient-evidence preservation (D-10 / RET-04): the merge stays
               insufficient ONLY when THIS attempt was insufficient AND the
               accumulated set adds nothing. If accumulated has notes (i.e. attempt
               1 found something), the merged result is sufficient by virtue of
               having any notes at all. If both empty, propagate True so the
               caller routes to INSUFFICIENT_EVIDENCE_MSG.
               """
               by_path: dict[str, RetrievedNote] = {}
               for note in accumulated:
                   by_path[note.path] = note
               for note in current.notes:
                   by_path[note.path] = note  # newer wins
               merged_notes = list(by_path.values())
               context_text = self._retrieval._assemble_context(merged_notes)
               merged_insufficient = (
                   current.insufficient_evidence and len(merged_notes) == 0
               )
               return RetrievalResult(
                   ok=True,
                   notes=tuple(merged_notes),
                   context_text=context_text,
                   insufficient_evidence=merged_insufficient,
               )

           def _build_trace(
               self,
               intent: str,
               attempts: list[RetrievalAttempt],
               final_ok: bool,
               t_start: float,
           ) -> RetrievalTrace:
               return RetrievalTrace(
                   enabled=True,
                   intent=intent,
                   attempts=tuple(attempts),
                   final_verdict_ok=final_ok,
                   judge_used=self._judge_enabled,
                   total_duration_ms=(time.monotonic() - t_start) * 1000.0,
               )


       __all__ = ["IterativeRetrieval", "RetrieveFn", "_too_similar"]
